74      private List<File> scanDir(FileMatcher fileMatcher, File dir) {
75          List<File> fileMatches = new ArrayList<>();
76  
77          File[] files = dir.listFiles();
78          if (files == null) {
79              throw new IllegalArgumentException("The abstract pathname of '" + dir +
80                      "' does not denote a directory, " +
81                      "or an I/O error occurs.");
82          }
83          for (File file : files) {
84  
85              if (file.isFile() && fileMatcher.isFileMatch(file)) {
86                  fileMatches.add(file);
87              } else if (file.isDirectory() && fileMatcher.isDirMatch(file)) {
88                  fileMatches.add(file);
89              }
90  
91              if (file.isDirectory() && fileMatcher.processDirectory(file)) {
92                  List<File> matchingSubFiles = scanDir(fileMatcher, file);
93                  fileMatches.addAll(matchingSubFiles);
94              }
95          }
96  
97          return fileMatches;
98      }

114     private List<PathMatcher> toPathMatchers(List<String> patterns) {
