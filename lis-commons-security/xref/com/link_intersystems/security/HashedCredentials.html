61      private static byte[] hash(MessageDigest messageDigest, char[] username, char[] password) {
62          username = username == null ? EMPTY_CHARS : username;
63          password = password == null ? EMPTY_CHARS : password;
64  
65          char[] credentials = new char[username.length + password.length];
66  
67          arraycopy(username, 0, credentials, 0, username.length);
68          fill(username, ' ');
69  
70          arraycopy(password, 0, credentials, username.length, password.length);
71          fill(password, ' ');
72  
73          CharBuffer charBuffer = CharBuffer.wrap(credentials);
74          ByteBuffer charsetEncodedCredentials = StandardCharsets.UTF_8.encode(charBuffer);
75          fill(credentials, ' ');
76  
77          byte[] charsetEncodedCredentialsArray = charsetEncodedCredentials.array();
78          int remaining = charsetEncodedCredentials.remaining();
79  
80          messageDigest.update(charsetEncodedCredentialsArray, 0, remaining);
81          byte[] hashedCredentials = messageDigest.digest();
82  
83          fill(charsetEncodedCredentialsArray, (byte) 0);
84  
85          return hashedCredentials;
86      }
