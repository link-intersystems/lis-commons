87      public Object clobToString(ResultSet resultSet, int columnIndex) throws SQLException {
88          StringBuffer stringBuffer = new StringBuffer();
89  
90          Clob clob = resultSet.getClob(columnIndex);
91          try (Reader reader = new BufferedReader(new InputStreamReader(clob.getAsciiStream()))) {
92              char[] buff = new char[8192];
93  
94              int read;
95              while ((read = reader.read(buff)) > 0) {
96                  stringBuffer.append(buff, 0, read);
97              }
98          } catch (IOException e) {
99              throw new SQLException(e);
100         }
