173     private HttpResponse executeRequest(HttpRequest httpRequest, HttpHeaders headers, ContentWriter contentWriter) throws IOException {
174         httpRequest.setHeaders(headers);
175 
176         requestCustomizer.customize(httpRequest);
177 
178         PreparedRequest preparedRequest = httpRequest.prepare();
179 
180         if (contentWriter != null) {

182             contentWriter.write(outputStream);
183         }
