16  package com.link_intersystems.lang.reflect;
17  
18  import com.link_intersystems.lang.ref.Reference;
19  
20  import java.lang.reflect.InvocationHandler;
21  import java.lang.reflect.Method;
22  import java.lang.reflect.Proxy;
23  import java.util.List;
24  
25  import static java.util.Objects.requireNonNull;
26  
27  /**
28   * A {@link DynamicDelegateProxy} is a {@link Proxy} that delegates calls to
29   * it's interface to a selected target object with the same interface. Target
30   * objects are obtained using a {@link Reference} and therefore can change at
31   * runtime. A {@link DynamicDelegateProxy} is useful when the target object must
32   * be determined dynamically or rather changes at runtime.
33   * <p>
34   * <strong>warning:</strong>
35   * <p>
36   * It's your responsibility to ensure that the {@link DynamicDelegateProxy}'s
37   * interface behaves in a consistent way. In some cases it might be hard to
38   * ensure consistency between calls, e.g. you should not replace a {@link List}
39   * proxy's target after a client called {@link List#size()} because the new
40   * target's size might be different and {@link IndexOutOfBoundsException}s might
41   * be the result.
42   * </p>
43   *
44   * @author Ren√© Link <a
45   * href="mailto:rene.link@link-intersystems.com">[rene.link@link-
46   * intersystems.com]</a>
47   * @since 1.2.0;
48   */
49  public class DynamicDelegateProxy<T> implements InvocationHandler {
50  
51      /**
52       * Creates an {@link DynamicDelegateProxy} that delegates calls to it's
53       * interface to a target object with the same interface.
54       *
55       * @param delegateClass proxy interface.
56       * @param targetRef     the targetRef that is used to obtain a target object when
57       *                      delegating a method invocation. The {@link Reference} must not
58       *                      return <code>null</code>.
59       * @return an {@link DynamicDelegateProxy} that delegates calls to it's
60       * interface to a target object with the same interface.
61       * @since 1.2.0;
62       */
63      @SuppressWarnings("unchecked")
64      public static <T> T create(Class<T> delegateClass,
65                                 Reference<? extends T> targetRef) {
66          requireNonNull(delegateClass);
67          requireNonNull(targetRef);
68  
69          Class<?>[] allInterfacesAsArray = new Class<?>[]{delegateClass};
70  
71          ClassLoader classLoader = delegateClass.getClassLoader();
72          DynamicDelegateProxy<T> delegateProxy = new DynamicDelegateProxy<>(
73                  targetRef);
74  
75          T delegate = (T) Proxy.newProxyInstance(classLoader,
76                  allInterfacesAsArray, delegateProxy);
77          return delegate;
78      }
79  
80      private Reference<? extends T> targetRef;
81  
82      protected DynamicDelegateProxy(Reference<? extends T> targetRef) {
83          requireNonNull(targetRef);
84          this.targetRef = targetRef;
85      }
86  
87      public Object invoke(Object proxy, Method method, Object[] args)
88              throws Throwable {
89          T target = targetRef.get();
90          Object result = method.invoke(target, args);
91          return result;
92      }
93  
94  }
