16  package com.link_intersystems.lang.reflect;
17  
18  import com.link_intersystems.lang.ref.AbstractSerializableReference;
19  import com.link_intersystems.lang.ref.Reference;
20  
21  import java.io.Serializable;
22  import java.lang.reflect.Field;
23  import java.lang.reflect.Modifier;
24  
25  import static java.util.Objects.requireNonNull;
26  
27  /**
28   * Represents a {@link Field} object {@link Reference} that is
29   * {@link Serializable}.
30   *
31   * @author Ren√© Link <a
32   * href="mailto:rene.link@link-intersystems.com">[rene.link@link-
33   * intersystems.com]</a>
34   * @since 1.0.0;
35   */
36  class SerializableField extends AbstractSerializableReference<Field> {
37  
38      /**
39       *
40       */
41      private static final long serialVersionUID = -6636456646515992155L;
42  
43      public SerializableField(Field field) {
44          super(requireNonNull(field));
45      }
46  
47      @Override
48      protected Serializable serialize(Field nonSerializableObject) {
49          return new MemberSerialization<Field>(nonSerializableObject);
50      }
51  
52      @SuppressWarnings("unchecked")
53      @Override
54      protected Field deserialize(Serializable restoreInfo) throws NoSuchFieldException {
55          MemberSerialization<Field> memberSerializationInfo = (MemberSerialization<Field>) restoreInfo;
56          Class<?> declaringClass = memberSerializationInfo.getDeclaringClass();
57          String fieldName = memberSerializationInfo.getMemberName();
58          int modifiers = memberSerializationInfo.getModifiers();
59          Field field = declaringClass.getDeclaredField(fieldName);
60          int currentModifiers = getModifier(field);
61          if (modifiers != currentModifiers) {
62              throw new IllegalStateException(
63                      "Unable to restore field "
64                              + fieldName
65                              + " declared at "
66                              + declaringClass
67                              + ". Modifiers changed since serialization. Expected modifiers are "
68                              + Modifier.toString(modifiers)
69                              + ", but current modifiers are "
70                              + Modifier.toString(currentModifiers));
71          }
72          return field;
73      }
74  
75      int getModifier(Field field) {
76          int currentModifiers = field.getModifiers();
77          return currentModifiers;
78      }
79  
80  }
