16  /**
17   * Creates a proxy that delegates all invocations to a delegate as long as it is
18   * not garbage collected. If the target has been garbage collected the proxy
19   * will return <code>null</code> for all non-primitive types and the
20   * {@link Primitives#getDefaultValue(Class)} in case of a primitive return type.
21   *
22   * <p/>
23   *
24   * A {@link WeakDelegateProxy} is most likely used for interfaces that do not
25   * return values like listeners. You can then implement an automatic listener
26   * deregistration.
27   *
28   * <pre>
29   * public class SomeObservable implements {@link ChangeListener} {
30  
31  	private List<ChangeListener> listeners = new ArrayList<>();
32  
33  	private String aPropery;
34  
35  	private void notifyChanged() {
36  	    // make a copy, because the original list might be changed while iterating, because a listener might get removed.
37  		List<ChangeListener> toNotify = new ArrayList<>(listeners);
38  		ChangeEvent event = new {@link ChangeEvent}(this);
39  
40  		// listeners that do not have a target anymore will be removed automatically
41  		toNotify.forEach(o -> o.stateChanged(event));
42  	}
43  
44  	&#64;Override
45  	public void addChangeListener({@link ChangeListener} listener) {
46  		WeakDelegateProxy<ChangeListener> listenerProxy = new WeakDelegateProxy<>(ChangeListener.class, listener, listeners::remove);
47  		listeners.add(listenerProxy);
48  	}
49  }
50   * </pre>
51   *
52   * You can also use the {@link Runnable} interface to clean up, if the proxies
53   * are created with that interface. See
54   * {@link #setRunnableProxyEnabled(boolean)}.
55   *
56   * <pre>
57   * private List<ChangeListener> listeners = new ArrayList<>();
58   *
59  	private void notifyChanged() {
60  	    for(ChangeListener listenerProxy = listeners){
61  	       if(listenerProxy instanceof Runnable){
62  	           Runnable runnable = (Runnable) runnable;
63  	           runnable.run();
64  	       }
65  	    }
66  		ChangeEvent event = new {@link ChangeEvent}(this);
67  		listeners.forEach(o -> o.stateChanged(event));
68  	}
69   * </pre>
70   *
71   * <b>note:</b> The {@link Runnable} interface was choosen, because it is always
72   * visible by the {@link ClassLoader} the proxy is created with.
73   *

75   *         <a href="mailto:rene.link@link-intersystems.com">[rene.link@link-
76   *         intersystems.com]</a>
77   * @param <T> the reference object's referent type.
78   *

81  public class WeakDelegateProxy<T> extends AbstractInvocationHandler<T> {

86  	private Consumer<T> onDelegateRemoved;

88  	public WeakDelegateProxy(Class<T> type, T delegate, Consumer<T> onDelegateRemoved) {
89  		super(type);
90  
91  		if (onDelegateRemoved == null) {
92  			onDelegateRemoved = o -> {
93  			};
94  		}
95  
96  		this.onDelegateRemoved = onDelegateRemoved;
97  		delegateReference = new WeakReference<T>(requireNonNull(delegate));
98  	}
99  
100 	@Override
101 	public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
102 		if (Runnable.class.equals(method.getDeclaringClass())) {
103 			checkReferent(proxy);
104 			return null;
105 		}
106 
107 		T t = checkReferent(proxy);
108 
109 		if (t == null) {
110 			Class<?> returnType = method.getReturnType();
111 			if (Primitives.isPrimitive(returnType)) {
112 				return Primitives.getDefaultValue(returnType);
113 			}
114 			return null;
115 		} else {
116 			return method.invoke(t, args);
117 		}
118 	}
119 
120 	@SuppressWarnings("unchecked")
121 	private T checkReferent(Object proxy) {
122 		T t = delegateReference.get();
123 
124 		if (t == null) {
125 			onDelegateRemoved.accept((T) proxy);
126 		}
127 
128 		return t;
129 	}

131 	public Optional<T> getDelegate() {
132 		return Optional.ofNullable(delegateReference.get());
133 	}
