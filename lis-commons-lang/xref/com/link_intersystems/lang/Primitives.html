34      private static final Map<Class<?>, Object> DEFAULT_VALUES;

36      static {
37          HashMap<Class<?>, Object> defaultValues = new HashMap<>();
38          defaultValues.put(Integer.TYPE, 0);
39          defaultValues.put(Boolean.TYPE, false);
40          defaultValues.put(Long.TYPE, 0L);
41          defaultValues.put(Short.TYPE, (short) 0);
42          defaultValues.put(Float.TYPE, 0f);
43          defaultValues.put(Character.TYPE, '\u0000');
44          defaultValues.put(Byte.TYPE, (byte) 0);
45          defaultValues.put(Double.TYPE, 0d);

49      public static final Map<Class<?>, Class<?>> primitiveToWrapper = new HashMap<>();
50  
51      static {
52          primitiveToWrapper.put(boolean.class, Boolean.class);
53          primitiveToWrapper.put(byte.class, Byte.class);
54          primitiveToWrapper.put(short.class, Short.class);
55          primitiveToWrapper.put(char.class, Character.class);
56          primitiveToWrapper.put(int.class, Integer.class);
57          primitiveToWrapper.put(long.class, Long.class);
58          primitiveToWrapper.put(float.class, Float.class);
59          primitiveToWrapper.put(double.class, Double.class);

62      public static Class<?> getWrapperType(Class<?> primitiveType) {
63          return primitiveToWrapper.get(primitiveType);
64      }

67      public static Class<?> getPrimitiveType(Class<?> primitiveWrapper) {
68          for (Map.Entry<Class<?>, Class<?>> primitiveToWrapper : primitiveToWrapper.entrySet()) {
69              if (primitiveToWrapper.getValue().equals(primitiveWrapper)) {
70                  return primitiveToWrapper.getKey();
71              }
72          }
73          return null;
74      }

81      /**
82       * @param type
83       * @return the corresponding autoboxing type of parameter type. Returns
84       * null, if parameter type is neither a primitive nor a primitive
85       * wrapper type. If the parameter is already an autoboxing type
86       * (e.g. Integer) it is returned.
87       * @since 1.0.0;
88       */
89      public static Class<?> getAutoboxingType(Class<?> type) {
90          if (isPrimitive(type)) {
91              return Conversions.getBoxingConversion(type);
92          } else if (isPrimitiveWrapper(type)) {
93              return type;
94          } else {
95              return null;
96          }
97      }

105     public static boolean isPrimitive(Class<?> type) {
106         if (type == null) {
107             return false;
108         } else {
109             return type.isPrimitive();

119     public static boolean isPrimitiveWrapper(Class<?> type) {

123     public static boolean isWideningConversionAllowed(Class<?> type,
124                                                       Class<?> typeToConvertTo) {
125         Class<?> from = type;
126         Class<?> to = typeToConvertTo;
127 
128         if (!Conversions.isBoxingType(from) || !Conversions.isBoxingType(to)) {
129             return false;
130         }
131 
132         if (isPrimitiveWrapper(from)) {
133             from = Conversions.getUnboxingConversion(from);
134         }
135         if (isPrimitiveWrapper(to)) {
136             to = Conversions.getUnboxingConversion(to);
137         }
138 
139         return Conversions.isPrimitiveWidening(from, to);
140     }

201     /**
202      * Calls the appropriated {@link PrimitiveCallback} method for the
203      * primitiveWrapper object.
204      *
205      * @param primitiveWrapper  a primitive wrapper object like {@link Byte}, {@link Integer}
206      *                          and so on.
207      * @param primitiveCallback the callback to call for the primitive wrapper type.
208      * @since 1.2.0;
209      */
210     public static void primitiveCallback(Object primitiveWrapper, PrimitiveCallback primitiveCallback) {
211         Class<?> primitiveWrapperClass = primitiveWrapper.getClass();
212         if (!Primitives.isPrimitiveWrapper(primitiveWrapperClass)) {
213             throw new IllegalArgumentException(
214                     "Argument primitiveWrapper must be a primitive wrapper type");
215         }
216         if (Integer.class.equals(primitiveWrapperClass)) {
217             Integer primitiveWrapperCasted = (Integer) primitiveWrapper;
218             primitiveCallback.primitive(primitiveWrapperCasted);
219         } else if (Long.class.equals(primitiveWrapperClass)) {
220             Long primitiveWrapperCasted = (Long) primitiveWrapper;
221             primitiveCallback.primitive(primitiveWrapperCasted);
222         } else if (Double.class.equals(primitiveWrapperClass)) {
223             Double primitiveWrapperCasted = (Double) primitiveWrapper;
224             primitiveCallback.primitive(primitiveWrapperCasted);
225         } else if (Character.class.equals(primitiveWrapperClass)) {
226             Character primitiveWrapperCasted = (Character) primitiveWrapper;
227             primitiveCallback.primitive(primitiveWrapperCasted);
228         } else if (Boolean.class.equals(primitiveWrapperClass)) {
229             Boolean primitiveWrapperCasted = (Boolean) primitiveWrapper;
230             primitiveCallback.primitive(primitiveWrapperCasted);
231         } else if (Byte.class.equals(primitiveWrapperClass)) {
232             Byte primitiveWrapperCasted = (Byte) primitiveWrapper;
233             primitiveCallback.primitive(primitiveWrapperCasted);
234         } else if (Float.class.equals(primitiveWrapperClass)) {
235             Float primitiveWrapperCasted = (Float) primitiveWrapper;
236             primitiveCallback.primitive(primitiveWrapperCasted);
237         } else {
238             Short primitiveWrapperCasted = (Short) primitiveWrapper;
239             primitiveCallback.primitive(primitiveWrapperCasted);
240         }
241     }

243     /**
244      * Converts a primitive wrapper array to the appropriated primitive array
245      * according to the primitive wrapper array's component type.</br>
246      *
247      * <pre>
248      * Byte[] primitiveWrapperArray = new Byte[] { 1, 2, 3, 4 };
249      *
250      * byte[] primitiveArray = Primitives
251      * 		.wrapperToPrimitiveArray(primitiveWrapperArray);
252      * </pre>
253      *
254      * @param primitiveWrapperArray
255      * @return the appropriated primitive array for the primitive wrapper array.
256      * Casts must respect the primitive wrapper array's component type.
257      * Due to the fact that generic definitions do not allow primitive
258      * types the return type can not be constraint further.
259      * @since 1.2.0;
260      */
261     @SuppressWarnings("unchecked")
262     public static <T> T wrapperToPrimitiveArray(Object[] primitiveWrapperArray) {
263         Class<?> primitiveWrapperArrayClass = primitiveWrapperArray.getClass();
264         Class<?> primitiveWrapperComponentType = primitiveWrapperArrayClass
265                 .getComponentType();
266         boolean isPrimitiveWrapperArray = Primitives
267                 .isPrimitiveWrapper(primitiveWrapperComponentType);
268 
269         Assert.isTrue(isPrimitiveWrapperArray,
270                 "primitiveWrapperArray's component type is not a primitive: %",
271                 primitiveWrapperComponentType);
272 
273         Class<?> primitiveArrayComponentType = Conversions
274                 .getUnboxingConversion(primitiveWrapperComponentType);
275         Object primitiveArray = Array.newInstance(primitiveArrayComponentType,
276                 primitiveWrapperArray.length);
277         PrimitiveArrayCallback primitiveArrayCallback = new PrimitiveArrayCallback(
278                 primitiveArray);
279         primitiveArrayCallback.setAutoincrementEnabled(false);
280         for (int i = 0; i < primitiveWrapperArray.length; i++) {
281             Object primitiveWrapper = primitiveWrapperArray[i];
282             primitiveArrayCallback.setIndex(i);
283             primitiveCallback(primitiveWrapper, primitiveArrayCallback);
284         }
285         return (T) primitiveArray;
286     }

288     /**
289      * Converts a primitive array to the appropriated primitive wrapper array
290      * according to the primitive array's component type.</br>
291      *
292      * <pre>
293      * byte[] primitiveArray = new byte[] { 1, 2, 3, 4 };
294      *
295      * Byte[] primitiveArray = Primitives.primitiveToWrapperArray(primitiveArray);
296      * </pre>
297      *
298      * @param primitiveArray
299      * @return the appropriated primitive wrapper array for the primitive array.
300      * Casts must respect the primitive array's component type.
301      * @since 1.2.0;
302      */
303     @SuppressWarnings("unchecked")
304     public static <T> T[] primitiveToWrapperArray(Object primitiveArray) {
305         Class<?> primitiveArrayClass = primitiveArray.getClass();
306         if (!primitiveArrayClass.isArray()) {
307             throw new IllegalArgumentException(
308                     "primitiveArray must be a primitive array, but was "
309                             + primitiveArrayClass);
310         }
311         Class<?> primitiveComponentType = primitiveArrayClass
312                 .getComponentType();
313         if (!Primitives.isPrimitive(primitiveComponentType)) {
314             throw new IllegalArgumentException(
315                     "primitiveArray's component type is not a primitive type: "
316                             + primitiveComponentType);
317         }
318 
319         Class<?> wrapperType = Conversions
320                 .getBoxingConversion(primitiveComponentType);
321         int length = Array.getLength(primitiveArray);
322         Object[] wrapperArray = (Object[]) Array.newInstance(wrapperType,
323                 length);
324 
325         for (int i = 0; i < length; i++) {
326             Object primitiveObject = Array.get(primitiveArray, i);
327             wrapperArray[i] = primitiveObject;
328         }
329         return (T[]) wrapperArray;
330 
331     }
