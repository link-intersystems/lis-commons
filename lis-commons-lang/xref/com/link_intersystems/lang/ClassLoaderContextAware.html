16  package com.link_intersystems.lang;
17  
18  import java.lang.ref.WeakReference;
19  import java.util.Map;
20  import java.util.WeakHashMap;
21  import java.util.concurrent.Callable;
22  
23  /**
24   * A {@link ClassLoaderContextAware} encapsulated a thread's context class
25   * loader management. It ensures that the {@link ClassLoaderContextAware}'s
26   * class loader is available as the thread's context class loader when a
27   * {@link Callable}s, {@link Runnable} or reflection method calls is executed.
28   * See {@link ContextAware} for more details.
29   *
30   * @author Ren√© Link <a
31   *         href="mailto:rene.link@link-intersystems.com">[rene.link@link-
32   *         intersystems.com]</a>
33   *
34   * @since 1.2.0;
35   *
36   * @see ContextAware#createContextProxy(Object)
37   * @see ContextAware#runInContext(Callable)
38   * @see ContextAware#runInContext(Runnable)
39   * @see ContextAware#invokeInContext(Object, String, Object...)
40   */
41  public final class ClassLoaderContextAware extends ContextAware<ClassLoader> {
42  
43  	private static final int DEFAULT_CACHE_SIZE = 1024;
44  
45  	public static final ClassLoaderContextAware SYSTEM_CLASS_LOADER_CONTEXT = new ClassLoaderContextAware(
46  			ClassLoader.getSystemClassLoader());
47  
48  	/**
49  	 * Class loaders can also be garbage collected and therefore we use a weak
50  	 * hash map to prevent memory leaks.
51  	 */
52  	private static final Map<ClassLoader, ClassLoaderContextAware> CLASS_LOADER_CONTEXT_CACHE = new WeakHashMap<ClassLoader, ClassLoaderContextAware>(
53  			DEFAULT_CACHE_SIZE);
54  
55  	private final ClassLoader contextClassLoader;
56  
57  	/**
58  	 * Creates a class loader context based on the given class loader.
59  	 *
60  	 * @param contextClassLoader
61  	 */
62  	private ClassLoaderContextAware(ClassLoader contextClassLoader) {
63  		this.contextClassLoader = contextClassLoader;
64  	}
65  
66  	/**
67  	 * Creates a {@link ClassLoaderContextAware} for the given classLoader. The
68  	 * system class loader is used if the given classLoader is <code>null</code>
69  	 * . This method uses an internal cache and returns the cached
70  	 * {@link ClassLoaderContextAware} if possible. The cache uses
71  	 * {@link WeakReference}s so that the {@link ClassLoaderContextAware}s will
72  	 * get garbage collected if they are only referenced by this cache.
73  	 *
74  	 * @param classLoader
75  	 *            the classLoader to use or <code>null</code> if the system
76  	 *            class loader should be used.
77  	 * @return
78  	 * @since 1.2.0;
79  	 */
80  	public static ClassLoaderContextAware forClassLoader(ClassLoader classLoader) {
81  		if (classLoader == null) {
82  			return SYSTEM_CLASS_LOADER_CONTEXT;
83  		}
84  
85  		ClassLoaderContextAware cachedCLassLoaderContext = getCachedClassLoaderContext(classLoader);
86  		if (cachedCLassLoaderContext != null) {
87  			return cachedCLassLoaderContext;
88  		}
89  
90  		ClassLoaderContextAware classLoaderContext = createAndCacheContextClassLoaderFor(classLoader);
91  
92  		return classLoaderContext;
93  	}
94  
95  	private static ClassLoaderContextAware createAndCacheContextClassLoaderFor(
96  			ClassLoader classLoader) {
97  		ClassLoaderContextAware classLoaderContext = new ClassLoaderContextAware(
98  				classLoader);
99  		CLASS_LOADER_CONTEXT_CACHE.put(classLoader, classLoaderContext);
100 		return classLoaderContext;
101 	}
102 
103 	private static ClassLoaderContextAware getCachedClassLoaderContext(
104 			ClassLoader classLoader) {
105 		return CLASS_LOADER_CONTEXT_CACHE.get(classLoader);
106 	}
107 
108 	/**
109 	 * Binds the class loader that this {@link ClassLoaderContextAware} was
110 	 * created with to the current threads
111 	 * {@link Thread#getContextClassLoader()}.
112 	 *
113 	 * @return the previous class loader that was bound to the current thread's
114 	 *         {@link Thread#getContextClassLoader()} when this method was
115 	 *         called.
116 	 * @since 1.2.0;
117 	 */
118 	@Override
119 	protected ClassLoader activateContext() {
120 		Thread currentThread = Thread.currentThread();
121 		ClassLoader initialClassLoader = currentThread.getContextClassLoader();
122 		currentThread.setContextClassLoader(contextClassLoader);
123 		return initialClassLoader;
124 	}
125 
126 	/**
127 	 * Unbinds this {@link ClassLoaderContextAware}'s class loader from this
128 	 * {@link Thread#getContextClassLoader()} and restores the class loader that
129 	 * was previously.
130 	 *
131 	 * @since 1.2.0;
132 	 */
133 	@Override
134 	protected void deactivateContext(ClassLoader previousClassLoader) {
135 		Thread currentThread = Thread.currentThread();
136 		currentThread.setContextClassLoader(previousClassLoader);
137 	}
138 
139 }
