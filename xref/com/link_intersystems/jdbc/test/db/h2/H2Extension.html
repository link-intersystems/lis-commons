14  public class H2Extension implements ParameterResolver, AfterTestExecutionCallback {

16      private H2DatabaseCache h2DatabaseStore;
17  
18      private H2DatabaseCache getH2DatabaseCache(ExtensionContext context) {
19          if (h2DatabaseStore == null) {
20              Optional<Class<?>> testClass = context.getTestClass();
21              h2DatabaseStore = testClass.map(c -> new H2DatabaseCache(new AnnotationH2ConfigPropertiesSource(c))).orElseThrow(() -> new IllegalStateException("No test class available"));
22          }
23          return h2DatabaseStore;
24      }
25  
26      @Override
27      public void afterTestExecution(ExtensionContext extensionContext) throws Exception {
28          JUnitExtensionH2DatabaseStore databaseStore = new JUnitExtensionH2DatabaseStore(extensionContext);
29          getH2DatabaseCache(extensionContext).removeDatabase(databaseStore);
30      }
31  
32      @Override
33      public boolean supportsParameter(ParameterContext parameterContext, ExtensionContext extensionContext) throws ParameterResolutionException {
34          Parameter parameter = parameterContext.getParameter();
35          Class<?> type = parameter.getType();
36          return Connection.class.equals(type) || H2Database.class.equals(type);
37      }
38  
39      @Override
40      public Object resolveParameter(ParameterContext parameterContext, ExtensionContext extensionContext) throws ParameterResolutionException {
41          if (supportsParameter(parameterContext, extensionContext)) {
42              Parameter parameter = parameterContext.getParameter();
43              Class<?> type = parameter.getType();
44  
45              try {
46                  H2DatabaseCache h2DatabaseCache = getH2DatabaseCache(extensionContext);
47                  JUnitExtensionH2DatabaseStore databaseStore = new JUnitExtensionH2DatabaseStore(extensionContext);
48                  H2Database h2Database = h2DatabaseCache.getDatabase(databaseStore);
49  
50                  if (type.equals(Connection.class)) {
51                      return h2Database.getConnection();
52                  } else {
53                      return h2Database;
54                  }
55              } catch (Exception e) {
56                  throw new ParameterResolutionException("Unable to open database connection", e);
57              }
58          }
59  
60          return null;
61      }
