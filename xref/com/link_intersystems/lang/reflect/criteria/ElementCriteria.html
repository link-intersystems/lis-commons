16  package com.link_intersystems.lang.reflect.criteria;
17  
18  import java.util.Iterator;
19  import java.util.function.Predicate;
20  
21  import static com.link_intersystems.util.Iterators.filtered;
22  import static java.util.Objects.requireNonNull;
23  
24  /**
25   * An {@link ElementCriteria} provides selection, and filtering capabilities for
26   * member based criteria. All{@link ElementCriteria}s provide methods to obtain an {@link Iterable}.
27   * After the {@link Iterable} is constructed it ensures that all {@link Iterator}s
28   * it constructs apply the same iteration rules. These are the rules of the critera at
29   * construction time of the {@link Iterable}. This rule is ensured for properties of the criteria.
30   * It can not be ensured for properties of dependent objects, e.g. for {@link Predicate}s that have state
31   * like a unique predicate and are added using {@link #add(Predicate)}.
32   *
33   * <p>
34   * <pre>
35   * ClassCriteria criteria = new ClassCriteria();
36   * criteria.setTraverseOrder(TraverseOrder.INTERFACES_FIRST);
37   * Iterator&lt;Class&lt;?&gt;&gt; classIterator = criteria.getIterable(ArrayList.class)
38   * 		.iterator();
39   * /*
40   *  * Should not have an effect on the iterator or iterable created before
41   *  &#42;/
42   * criteria.setTraverseOrder(TraverseOrder.SUPERCLASS_ONLY);
43   * criteria.add(new AssignablePredicate(AbstractList.class));
44   * criteria.setResultSelection(Selection.LAST);
45   *
46   * Iterator&lt;Class&lt;?&gt;&gt; secondIterator = criteria.getIterable(ArrayList.class)
47   * 		.iterator();
48   * </pre>
49   *
50   * </p>
51   *
52   * @author Ren√© Link <a
53   * href="mailto:rene.link@link-intersystems.com">[rene.link@link-
54   * intersystems.com]</a>
55   * @since 1.0.0;
56   */
57  public abstract class ElementCriteria<T> implements Cloneable {
58  
59      private Predicate<T> elementFilterPredicate = null;
60  
61      private Result select = Result.ALL;
62  
63      public ElementCriteria() {
64      }
65  
66      /**
67       * Use the defined predicate as a filter when searching for elements.
68       * Multiple calls to this method will combine the predicates with a logical
69       * 'and' operator. The type of objects that the {@link Predicate} must be
70       * able to handle is defined by subclasses.
71       *
72       * @param predicate the predicate to use when filtering elements.
73       * @since 1.0.0;
74       */
75      public void add(Predicate<T> predicate) {
76          if (elementFilterPredicate == null) {
77              elementFilterPredicate = requireNonNull(predicate);
78          } else {
79              elementFilterPredicate = elementFilterPredicate.and(requireNonNull(predicate));
80          }
81      }
82  
83      /**
84       * Takes the iterator and wraps it into a filtering iterator that applies
85       * the {@link Predicate}s that were configured via {@link #add(Predicate)}.
86       *
87       * @param iterator the iterator that the filter should be applied on.
88       * @return an {@link Iterator} that filters the elements of the given
89       * {@link Iterator} according to the filter {@link Predicate}s
90       * defined by {@link #add(Predicate)}.
91       * @since 1.0.0;
92       */
93      protected Iterator<T> applyElementFilter(Iterator<T> iterator) {
94          if (elementFilterPredicate == null) {
95              return iterator;
96          } else {
97              return filtered(iterator, elementFilterPredicate);
98          }
99      }
100 
101     /**
102      * Takes the iterator and wraps it into a filtering iterator that applies
103      * the selection requirements as they are configured via
104      * {@link #setResult(Result)}.
105      *
106      * @param iterator the iterator that the filter should be applied on.
107      * @return an {@link Iterator} that selects the elements of the given
108      * {@link Iterator} according to selection as specified by
109      * {@link #setResult(Result)}.
110      * @since 1.0.0;
111      */
112     protected Iterator<T> applySelectionFilter(final Iterator<T> iterator) {
113         return select.apply(iterator);
114     }
115 
116     /**
117      * Only select elements as described by the {@link Result} argument.
118      *
119      * @param select specifies which matched elements will be returned. Defaults to
120      *               {@link Result#ALL}
121      * @since 1.0.0;
122      */
123     public void setResult(Result select) {
124         this.select = requireNonNull(select);
125     }
126 
127     @Override
128     protected ElementCriteria<T> clone() {
129         try {
130             return (ElementCriteria<T>) super.clone();
131         } catch (CloneNotSupportedException e) {
132             throw new IllegalStateException(e);
133         }
134     }
135 }
