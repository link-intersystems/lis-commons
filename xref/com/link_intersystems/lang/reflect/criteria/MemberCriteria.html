34  /**
35   * Iterate over {@link Member}s or {@link AnnotatedElement}s that match the
36   * criteria specified by a {@link MemberCriteria}.
37   * <p>
38   * <h2>Examples</h2>
39   * <h3>Iterate over all public abstract methods, in method names order, whose
40   * name's match the pattern &quot;add.*&quot; starting at class ArrayList.</h3>
41   *
42   * <pre>
43   * MemberCriteria memberCriteria = new MemberCriteria();
44   * memberCriteria.withAccess(AccessType.PUBLIC);
45   * memberCriteria.withModifiers(Modifier.ABSTRACT);
46   * memberCriteria.membersOfType(Method.class);
47   * memberCriteria.setMemberIterateOrder(MemberNameComparator.INSTANCE);
48   * memberCriteria.named(Pattern.compile(&quot;add.*&quot;));
49   *
50   * ClassCriteria classCriteria = new ClassCriteria();
51   * Iterable&lt;Class&lt;?&gt;&gt; classIterable = classCriteria.getIterable(ArrayList.class);
52   * Iterable&lt;Member&gt; memberIterable = memberCriteria.getIterable(classIterable);
53   *
54   * Iterator&lt;Member&gt; criteriaIterator = memberIterable.iterator();
55   * </pre>
56   *
57   * <h3>Get the first public method that has the same signature as
58   * List.subList(int, int) starting at ArrayList traversing only the class
59   * hierarchy (omit interfaces).</h3>
60   *
61   * <pre>
62   * MemberCriteria memberCriteria = new MemberCriteria();
63   * memberCriteria.setResult(Selection.FIRST);
64   * memberCriteria.membersOfType(Method.class);
65   *
66   * Method method = List.class.getDeclaredMethod(&quot;subList&quot;, int.class, int.class);
67   *
68   * memberCriteria.add(new SignaturePredicate(method));
69   *
70   * ClassCriteria classCriteria = new ClassCriteria();
71   * classCriteria.setTraverseOrder(TraverseOrder.SUPERCLASSES_ONLY);
72   * Iterable&lt;Class&lt;?&gt;&gt; classIterable = classCriteria.getIterable(ArrayList.class);
73   * Iterable&lt;Member&gt; memberIterable = memberCriteria.getIterable(classIterable);
74   *
75   * Iterator&lt;Member&gt; criteriaIterator = memberIterable.iterator();
76   * Member member = criteriaIterator.next();
77   * Method firstMatch = (Method) member;
78   * Method declaredMethod = AbstractList.class.getDeclaredMethod(&quot;subList&quot;, int.class, int.class);
79   * assertEquals(declaredMethod, firstMatch);
80   * assertFalse(criteriaIterator.hasNext());
81   * </pre>
82   *
83   * </p>
84   *
85   * @author René Link
86   * <a href="mailto:rene.link@link-intersystems.com">[rene.link@link-
87   * intersystems.com]</a>
88   * @since 1.0.0;
89   */

160     /**
161      * Add a criterion to only select {@link Member}s of the specified types.
162      * Allowed types are
163      * <ul>
164      * <li>{@link Constructor}</li>
165      * <li>{@link Method}</li>
166      * <li>{@link Field}</li>
167      * </ul>
168      *
169      * @param memberTypes the member types that this {@link MemberCriteria} should
170      *                    select.
171      * @since 1.0.0;
172      */
173     public final void membersOfType(Class<?>... memberTypes) {
174         /*
175          * we do not enforce member types by the generic definition
176          * "Class<? extends Member> ... memberTypes" because this will cause
177          * type unsafe warnings at the client side. For example if a client
178          * calls this method with the parameters
179          * "membersOfType(Field.class, Method.class)".
180          */
181         for (Class<?> memberType : memberTypes) {
182             if (!(Constructor.class.equals(memberType)) && !(Method.class.equals(memberType)) && !(Field.class.equals(memberType))) {
183                 throw new IllegalArgumentException("Unsupported member type " + memberType + ". Supported types are " + DEFAULT_MEMBER_TYPES + ".");
184             }
185         }
186         this.memberTypes = Arrays.asList(memberTypes);
187     }
188 
189     protected List<Member> getSortedMembers(Class<?> currentClass) {
190         List<Member> memberList = new ArrayList<>();
191         for (Class<?> memberType : memberTypes) {
192             Member[] members = null;
193             if (Constructor.class.equals(memberType)) {
194                 members = currentClass.getDeclaredConstructors();
195             } else if (Method.class.equals(memberType)) {
196                 members = currentClass.getDeclaredMethods();
197             } else if (Field.class.equals(memberType)) {
198                 members = currentClass.getDeclaredFields();
199             }
200             if (members != null) {
201                 List<Member> asList = Arrays.asList(members);
202                 memberList.addAll(asList);
203             }
204         }
205 
206         memberList.sort(iterateOrderComparator);
207         return memberList;
208     }

323     @SuppressWarnings("unchecked")
324     protected Iterator<T> applyAccessAndNamePredicates(Iterator<T> iterator) {
325         List<Predicate<T>> predicates = new ArrayList<>();
326 
327         Collection<AccessType> accesses = getAccesses();
328         int accessModifiers = 0;
329         for (AccessType access : accesses) {
330             switch (access) {
331                 case PUBLIC:
332                     accessModifiers |= Modifier.PUBLIC;
333                     break;
334                 case PRIVATE:
335                     accessModifiers |= Modifier.PRIVATE;
336                     break;
337                 case PROTECTED:
338                     accessModifiers |= Modifier.PROTECTED;
339                     break;
340                 default:
341                     break;
342             }
343         }
344         Predicate<Member> accessModifierPredicate = new MemberModifierPredicate(accessModifiers, MemberModifierPredicate.Match.AT_LEAST_ONE);
345 
346         if (accesses.contains(AccessType.DEFAULT)) {
347             MemberModifierPredicate memberModifierPredicate = new MemberModifierPredicate(Modifier.PRIVATE | Modifier.PUBLIC | Modifier.PROTECTED, MemberModifierPredicate.Match.AT_LEAST_ONE);
348             accessModifierPredicate = accessModifierPredicate.or(o -> !memberModifierPredicate.test(o));
349         }
350 
351         predicates.add((Predicate<T>) accessModifierPredicate);
352 
353         int modifiers = getModifiers();
354         if (modifiers != 0) {
355             predicates.add((Predicate<T>) new MemberModifierPredicate(modifiers, MemberModifierPredicate.Match.AT_LEAST_ALL));
356         }
357 
358         String name = getName();
359         if (name != null) {
360             predicates.add((Predicate<T>) ReflectFacade.getMemberNamePredicate(name));
361         }
362 
363         Pattern pattern = getPattern();
364         if (pattern != null) {
365             predicates.add((Predicate<T>) ReflectFacade.getMemberNamePatternPredicate(pattern));
366         }
367         Predicate<T> allPredicate = andPredicate(predicates);
368         iterator = filtered(iterator, allPredicate);
369         return iterator;
370     }
371 
372     private Predicate<T> andPredicate(List<Predicate<T>> predicates) {
373         Predicate<T> result = null;
374 
375         for (Predicate<T> predicate : predicates) {
376             if (result == null) {
377                 result = predicate;
378             } else {
379                 result = result.and(predicate);
380             }
381         }
382 
383         return result;
384     }

566     private static class AnnotatedElementIteratorTransformer extends AbstractCriteriaTransformer<Object, Object> {
567 
568         private final IterateStrategy traverseStrategy;
569 
570         private final Collection<Class<?>> transformed = new ArrayList<>();
571 
572         public AnnotatedElementIteratorTransformer(IterateStrategy traverseStrategy, MemberCriteria memberCriteria) {
573             super(memberCriteria);
574             this.traverseStrategy = traverseStrategy;
575         }
576 
577         public Object apply(Object input) {
578             if (input instanceof Class<?>) {
579                 final Class<?> currentClass = (Class<?>) input;
580                 boolean hasBeenTransformed = transformed.remove(currentClass);
581                 if (hasBeenTransformed) {
582                     switch (traverseStrategy) {
583                         case MEMBERS_ONLY:
584                         case PACKAGES_ONLY:
585                             return Collections.emptyIterator();
586                         default:
587                             return currentClass;
588                     }
589                 }
590 
591                 transformed.add(currentClass);
592 
593                 MemberCriteria memberCriteria = getMemberCriteria();
594 
595                 JavaElementTraverseStrategy javaElementTraverseStrategy = traverseStrategy.getJavaElementTraverseStrategy();
596 
597                 Iterator<?> iterator = javaElementTraverseStrategy.getIterator(currentClass, memberCriteria);
598 
599                 return iterator;
600             } else {
601                 return input;
602             }
603         }
