12   * Convenience class to access constants defined in some class.
13   *
14   * @author Ren√© Link {@literal <rene.link@link-intersystems.com>}
15   */
16  public class Constants<T> extends AbstractList<T> {
17  
18      private Map<String, Field> byName;
19      private List<Field> constantFields;
20      private Map<Field, T> cachedConstantValues = new HashMap<>();
21  
22      public Constants(Class<T> contantType) {
23          this(contantType, contantType);
24      }
25  
26      public Constants(Class<?> constantDeclaringClass, Class<T> constantType) {
27          byName = Arrays.stream(constantDeclaringClass.getDeclaredFields())
28                  .filter(this::isPublicStaticFinal)
29                  .filter(f -> constantType.equals(f.getType()))
30                  .collect(toMap(
31                          Field::getName,
32                          f -> f,
33                          throwingMerger(),
34                          LinkedHashMap::new));
35  
36  
37          constantFields = new ArrayList<>(byName.values());
38      }
39  
40      private static <T> BinaryOperator<T> throwingMerger() {
41          return (u, v) -> {
42              throw new IllegalStateException(String.format("Duplicate key %s", u));
43          };
44      }
45  
46      private boolean isPublicStaticFinal(Member member) {
47          int m = member.getModifiers();
48          return isPublic(m) && isStatic(m) && isFinal(m);
49      }
50  
51      public T getValue(String name) {
52          Field constantField = byName.get(name);
53          return getConstantValue(constantField);
54      }
55  
56      @Override
57      public T get(int index) {
58          Field constantField = constantFields.get(index);
59          return getConstantValue(constantField);
60      }
61  
62      private T getConstantValue(Field constantField) {
63          if (constantField == null) {
64              return null;
65          }
66          return cachedConstantValues.computeIfAbsent(constantField, this::getFieldValue);
67      }
68  
69      @SuppressWarnings("unchecked")
70      private T getFieldValue(Field field) {
71          try {
72              return (T) field.get(field.getDeclaringClass());
73          } catch (IllegalAccessException e) {
74              throw new IllegalStateException(e);
75          }
76      }
77  
78      @Override
79      public int size() {
80          return constantFields.size();
81      }
82  }
