16  package com.link_intersystems.lang;
17  
18  import java.net.URL;
19  import java.net.URLClassLoader;
20  
21  /**
22   * A {@link URLClassLoader} that first tries to load a class by it's own and if
23   * it can not be loaded it asks it's parent class loader to load.
24   *
25   * @author Ren√© Link <a
26   * href="mailto:rene.link@link-intersystems.com">[rene.link@link-
27   * intersystems.com]</a>
28   * @since 1.2.0;
29   */
30  public class ParentLastURLClassLoader extends URLClassLoader {
31  
32      private ClassLoader parent;
33  
34      /**
35       * Constructs a new ParentLastURLClassLoader for the specified URLs using
36       * this class loader's {@link ClassLoader} as it's parent . The URLs will be
37       * searched in the order specified for classes and resources after first
38       * searching in the parent class loader. Any URL that ends with a '/' is
39       * assumed to refer to a directory. Otherwise, the URL is assumed to refer
40       * to a JAR file which will be downloaded and opened as needed.
41       *
42       * <p>
43       * If there is a security manager, this method first calls the security
44       * manager's <code>checkCreateClassLoader</code> method to ensure creation
45       * of a class loader is allowed.
46       *
47       * @param urls the URLs from which to load classes and resources
48       * @throws SecurityException if a security manager exists and its
49       *                           <code>checkCreateClassLoader</code> method doesn't allow
50       *                           creation of a class loader.
51       * @see SecurityManager#checkCreateClassLoader
52       * @since 1.2.0;
53       */
54      public ParentLastURLClassLoader(URL[] urls) {
55          this(urls, ParentLastURLClassLoader.class.getClassLoader());
56      }
57  
58      /**
59       * Same as {@link #ParentLastURLClassLoader(URL[])}, but uses the given
60       * <code>parent</code> as it's parent class loader.
61       *
62       * @param urls   the URLs from which to load classes and resources
63       * @param parent the parent class loader to use. Can be <code>null</code>.
64       * @throws SecurityException if a security manager exists and its
65       *                           <code>checkCreateClassLoader</code> method doesn't allow
66       *                           creation of a class loader.
67       * @since 1.2.0;
68       */
69      public ParentLastURLClassLoader(URL[] urls, ClassLoader parent) {
70          super(urls);
71          this.parent = parent;
72      }
73  
74      /**
75       * Tries to load the requested class by it's own and delegates to it's
76       * parent if itself can not load the class.
77       * <p>
78       * {@inheritDoc}
79       * </p>
80       *
81       * @since 1.2.0;
82       */
83      protected synchronized Class<?> loadClass(String name, boolean resolve)
84              throws ClassNotFoundException {
85          // First, check if the class has already been loaded
86          Class<?> c = findLoadedClass(name);
87          if (c == null) {
88              try {
89                  c = findClass(name);
90              } catch (ClassNotFoundException e) {
91                  if (parent != null) {
92                      c = parent.loadClass(name);
93                  } else {
94                      throw new ClassNotFoundException(name, e);
95                  }
96              }
97          }
98          if (resolve) {
99              resolveClass(c);
100         }
101         return c;
102     }
