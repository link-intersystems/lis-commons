11  public class CharSequenceDetector {

19      public CharSequenceDetector(CharSequenceArrayFactory charSequenceArrayFactory) {
20          this.charSequenceArrayFactory = requireNonNull(charSequenceArrayFactory);
21      }
22  
23      /**
24       * @param pushbackReader the pushbackReader to use to detect the char sequence. The {@link PushbackReader} must
25       *                       have at least a pushback buffer size of 1.
26       * @return true if the char sequence was detected or false otherwise. If the char sequence was detected all
27       * chars that have been read during detection are gone and the {@link PushbackReader} is at the end position,
28       * so that it will return the next char after the detected char sequence on the next read call.
29       */
30      public boolean detect(PushbackReader pushbackReader, CharSequence sequence) throws IOException {
31  
32          if (sequence.length() == 0) {
33              return !hasMoreChars(pushbackReader);
34          }
35  
36          int matchLength = getSequenceMatchLength(pushbackReader, sequence);
37  
38          boolean completeMatch = sequence.length() == matchLength;
39          if (!completeMatch) {
40              char[] matchedSequence = getMatchedSequence(sequence, matchLength);
41              pushbackReader.unread(matchedSequence);
42          }
43          return completeMatch;
44      }
45  
46      protected char[] getMatchedSequence(CharSequence sequence, int matchLength) {
47          return charSequenceArrayFactory.toArray(sequence, 0, matchLength);
48      }
49  
50      protected int getSequenceMatchLength(PushbackReader pushbackReader, CharSequence sequence) throws IOException {
51          int matchPosition = 0;
52  
53          int read;
54          while (matchPosition < sequence.length() &&
55                  (read = pushbackReader.read()) != -1) {
56  
57              char charAt = sequence.charAt(matchPosition);
58  
59              if (charAt != read) {
60                  pushbackReader.unread(read);
61                  break;
62              }
63  
64              matchPosition++;
65          }
66  
67          return matchPosition;
68      }
69  
70      protected boolean hasMoreChars(PushbackReader pushbackReader) throws IOException {
71          int read = pushbackReader.read();
72          if (read == -1) {
73              return false;
74          }
75          pushbackReader.unread(read);
76          return true;
77      }
