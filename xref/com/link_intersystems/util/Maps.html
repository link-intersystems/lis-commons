16  package com.link_intersystems.util;
17  
18  import java.util.Collection;
19  import java.util.HashMap;
20  import java.util.Map;
21  import java.util.function.Function;
22  
23  /**
24   * The {@link Maps} class provides {@link Map} related utility methods.
25   *
26   * @author - Ren√© Link {@literal <rene.link@link-intersystems.com>}
27   * @since 1.3.0;
28   */
29  public abstract class Maps {
30  
31      private static final double HASH_MAP_LOAD_FACTOR = 1 / 0.75;
32  
33      /**
34       * Safely builds a map from a {@link Collection} or objects by preventing duplicate keys.
35       * The object keys are calculated via a {@link ParameterizedObjectFactory}. The
36       * {@link ParameterizedObjectFactory} must ensure that a distinct key for
37       * every object instance in the collection is generated. If there are two
38       * instances that have the same key a {@link KeyCollisionException} is thrown.
39       *
40       * @param <K>            the key's type
41       * @param <V>            the value's type
42       * @param objects        the {@link Collection} of objects the key map should be
43       *                       generated of.
44       * @param keyGenFunction the {@link ParameterizedObjectFactory} to use for creating the
45       *                       keys.
46       * @return a map that maps the keys generated by the
47       * {@link ParameterizedObjectFactory} to the objects of the original
48       * collection.
49       * @throws KeyCollisionException if the collection contains two objects that are equal in
50       *                               their key which is generated by the
51       *                               {@link ParameterizedObjectFactory}.
52       * @since 1.3.0;
53       */
54      public static <K, V> Map<K, V> keyMap(
55              Collection<V> objects,
56              Function<V, K> keyGenFunction)
57              throws KeyCollisionException {
58          Map<K, V> map = new HashMap<>(
59                  (int) (objects.size() * HASH_MAP_LOAD_FACTOR));
60          for (V v : objects) {
61              K k = keyGenFunction.apply(v);
62              if (map.containsKey(k)) {
63                  throw new KeyCollisionException(k);
64              } else {
65                  map.put(k, v);
66              }
67          }
68          return map;
69      }
70  }
