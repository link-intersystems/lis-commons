6    * Provides utility methods to adapt {@link Function}, {@link Consumer}, {@link BiConsumer} and {@link Runnable}s
7    * to {@link FunctionExecutor#exec(Runnable)} calls.

9   public class FunctionExecutorAdapters {

13      public FunctionExecutorAdapters(FunctionExecutor functionExecutor) {
14          this.functionExecutor = functionExecutor;
15      }

17      public <T, R> Function<T, R> adapter(Function<T, R> function) {
18  
19          return t -> {
20              FunctionRunnable<T, R> functionRunnable = new FunctionRunnable<>(function, t);
21              functionExecutor.exec(functionRunnable);
22              return functionRunnable.getResult();
23          };
24      }
25  
26      public <T, U, R> BiFunction<T, U, R> adapter(BiFunction<T, U, R> function) {
27  
28          return (t, u) -> {
29              BiFunctionRunnable<T, U, R> functionRunnable = new BiFunctionRunnable<>(function, t, u);
30              functionExecutor.exec(functionRunnable);
31              return functionRunnable.getResult();
32          };
33      }
34  
35      public <T, U> BiConsumer<T, U> adapter(BiConsumer<T, U> consumer) {
36  
37          return (t, u) -> functionExecutor.exec(consumer, t, u);
38      }
39  
40      public <T, U> Consumer<T> adapter(BiConsumer<T, U> consumer, U arg) {

42  
43      }
44  
45      public <T, U> Consumer<T> adapter(BiConsumer<T, U> consumer, Supplier<U> argSupplier) {

48      }
49  
50      public <T> Consumer<T> adapter(Consumer<T> consumer) {
51
