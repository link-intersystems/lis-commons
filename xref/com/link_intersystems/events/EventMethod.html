15  /**
16   * An {@link EventMethod} is an adapter that can adapt method references,
17   * such as {@link Runnable}, {@link Consumer} and {@link BiConsumer}, to event listener methods.
18   *
19   * <pre>
20   * public class EventMethodUsageExample {
21   *
22   *     public static void main(String[] args) {
23   *         EventMethodUsageExample eventMethodUsageExample = new EventMethodUsageExample();
24   *         eventMethodUsageExample.setSelection(3);
25   *         eventMethodUsageExample.setSelection(5);
26   *     }
27   *
28   *     private ListSelectionListener l1 = ListSelectionEventMethod.VALUE_CHANGED.listener(this::printEventFired);
29   *
30   *     private ListSelectionListener l2 = ListSelectionEventMethod.VALUE_CHANGED.listener(
31   *             this::printSelectionChanged,
32   *             ListSelectionEvent::getLastIndex,
33   *             ">>> ", e -> !e.getValueIsAdjusting());
34   *
35   *     private ListSelectionModel selectionModel = new DefaultListSelectionModel();
36   *
37   *     EventMethodUsageExample() {
38   *         selectionModel.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
39   *         selectionModel.addListSelectionListener(l1);
40   *         selectionModel.addListSelectionListener(l2);
41   *     }
42   *
43   *     public void setSelection(int index) {
44   *         selectionModel.setValueIsAdjusting(true);
45   *         selectionModel.setSelectionInterval(index, index);
46   *         selectionModel.setValueIsAdjusting(false);
47   *     }
48   *
49   *     private void printEventFired(ListSelectionEvent e) {
50   *         int firstIndex = e.getFirstIndex();
51   *         int lastIndex = e.getLastIndex();
52   *         boolean isAdjusting = e.getValueIsAdjusting();
53   *         System.out.println("Selection model event fired:" +
54   *                 " firstIndex= " + firstIndex +
55   *                 " lastIndex= " + lastIndex +
56   *                 " isAdjusting= " + isAdjusting);
57   *     }
58   *
59   *     private void printSelectionChanged(int selectedIndex, String indent) {
60   *         System.out.println(indent + "Index " + selectedIndex + " selected");
61   *     }
62   * }
63   * </pre>
64   * <p>
65   * Running the example code above will output:
66   *
67   * <pre>
68   * Selection model event fired: firstIndex= 3 lastIndex= 3 isAdjusting= true
69   * >>> Index 3 selected
70   * Selection model event fired: firstIndex= 3 lastIndex= 3 isAdjusting= false
71   * Selection model event fired: firstIndex= 3 lastIndex= 5 isAdjusting= true
72   * >>> Index 5 selected
73   * Selection model event fired: firstIndex= 3 lastIndex= 5 isAdjusting= false
74   * </pre>
75   *
76   * @param <L> the event listener type.
77   * @param <E> the event object type.
78   */

90      /**
91       * Creates an {@link EventMethod} which supports the given event methods (by name) and whose event listener class
92       * and event object class are resolved by the generic type arguments that a subclass defines . E.g.
93       *
94       * <pre>
95       * public class ActionEventMethod extends EventMethod&lt;ActionListener, ActionEvent> {
96       *
97       *     public static final ActionEventMethod PERFORMED = new ActionEventMethod("actionPerformed");
98       *
99       *     public ActionEventMethod(String... methodNames) {
100      *         super(methodNames);
101      *     }
102      * }
103      * </pre>
104      * <p>
105      * The resolution algorithm also searches superclasses. E.g.
106      *
107      * <pre>
108      *     abstract class BaseEventMethod extends EventMethod&lt;SomeListener, SomeEvent> {
109      *     }
110      *
111      *     class ExtendsBase extends BaseEventMethod {
112      *     }
113      * </pre>
114      * <p>
115      * But the search algorithm has limitations. E.g. you can not split the type definition in the hierarchy. E.g.
116      *
117      * <pre>
118      *     abstract class BaseEventMethod<L> extends EventMethod<L, SomeEvent> {
119      *     }
120      *
121      *     class ExtendsBase extends BaseEventMethod&lt;SomeListener> {
122      *     }
123      * </pre>
124      * <p>
125      * and you may also not swap the type parameter positions.
126      *
127      * <pre>
128      *     abstract class BaseEventMethod&lt;E extends EventObject, L> extends EventMethod<L, E> {
129      *     }
130      *
131      *     class ExtendsBase extends BaseEventMethod&lt;SomeEvent, SomeListener> {
132      *     }
133      * </pre>
134      * <p>
135      * However, you can pass the arguments through multiple hierarchy levels.
136      *
137      * <pre>
138      *     abstract class BaseEventMethod&lt;L, E extends EventObject> extends EventMethod&lt;L, E> {
139      *     }
140      *
141      *     class ExtendsBase extends BaseEventMethod&lt;SomeListener, SomeEvent> {
142      *     }
143      * </pre>
144      * <p>
145      * or you can simply create an anonymous class if you don't want to create a custom subclass.
146      *
147      * <pre>
148      *     EventMethod eventMethod = new EventMethod&lt;SomeListener, SomeEvent>(){};
149      * </pre>
150      *
151      * @param methodNames the method names that this {@link EventMethod} should support.
152      */
153     protected EventMethod(String... methodNames) {
154         this.listenerClassSupplier = () -> resolveGenericClass(getClass(), 0);
155         this.eventObjectClassSupplier = () -> resolveGenericClass(getClass(), 1);
156         this.eventMethodsSupplier = new EventMethodResolver(methodNames, this::getListenerClass, this::getEventObjectClass);
157     }

196     /**
197      * A joined {@link EventMethod} is an {@link EventMethod} that supports this {@link EventMethod}'s methods and
198      * the {@link EventMethod} methods of the other {@link EventMethod}.
199      * <p>
200      * Assume you have defined an {@link EventMethod} that supports the mouse clicked event.
201      * <pre>
202      *   EventMethod&lthMouseListener,MouseEvent> clickedEvent = new EventMethod&lt;>(MouseListener.class, MouseEvent.class, "mouseClicked");
203      * </pre>
204      * <p>
205      * and an EventMethod that supports the mouse pressed event.
206      *
207      * <pre>
208      *   EventMethod&lthMouseListener,MouseEvent> pressedEvent = new EventMethod&lt;>(MouseListener.class, MouseEvent.class, "mousePressed");
209      * </pre>
210      * <p>
211      * You can then create a new {@link EventMethod} that supports both event methods by joining them.
212      * <pre>
213      *   EventMethod&lthMouseListener,MouseEvent> clickedAndPressedEvents = clickedEvent.join(pressedEvent);
214      * </pre>
215      * </p>
216      *
217      * @param otherEventMethod the other event method to join with.
218      * @return a new {@link EventMethod} instance that supports all event methods of this and the otherEventMethod.
219      * @throws IllegalArgumentException if the otherEventMethod is incompatible with this event and can therefore not
220      *                                  be joined. The otherEventMethod is incompatible if the event listener class or
221      *                                  the event object class are not equal.
222      * @see #isCompatible(EventMethod)
223      */
224     public EventMethod<L, E> join(EventMethod<L, E> otherEventMethod) {
225         if (!isCompatible(otherEventMethod)) {
226             Class<?> otherListenerClass = otherEventMethod.getListenerClass();
227             Class<?> eventObjectClass = this.getEventObjectClass();
228             Class<?> thisListenerClass = getListenerClass();
229             Class<?> otherEventObjectClass = otherEventMethod.getEventObjectClass();
230 
231             String msg = format("EventMethod can not be joined, because it's event listener and event object type is incompatible." +
232                             " {0}.{1} != {2}.{3}", thisListenerClass.getSimpleName(),
233                     eventObjectClass.getSimpleName(),
234                     otherListenerClass.getSimpleName(),
235                     otherEventObjectClass.getSimpleName());
236             throw new IllegalArgumentException(msg);
237         }
238 
239         List<Method> thisEventMethods = getEventMethods();
240         LinkedHashSet<Method> joinMethodsUnique = new LinkedHashSet<>(thisEventMethods);
241 
242         List<Method> otherEventMethods = otherEventMethod.getEventMethods();
243         joinMethodsUnique.addAll(otherEventMethods);
244 
245         List<Method> joinedMethods = new ArrayList<>(joinMethodsUnique);
246 
247         return new EventMethod<>(this, joinedMethods);
248     }

258         if (listenerClass == null) {
259             listenerClass = listenerClassSupplier.get();
260 
261             if (listenerClass == null) {
262                 String msg = "Can not resolve listener class by the generic superclass of" +
263                         " any superclass in the hierarchy. Either override getListenerClass() or" +
264                         " use the constructor that has a listener class parameter.";
265                 throw new IllegalStateException(msg);

456     @SuppressWarnings("unchecked")
457     protected <T> Class<T> resolveGenericClass(Class<?> clazz, int genericTypeArgIndex) {
458         if (clazz == null) {
459             return null;
460         }
461 
462         Type genericSuperclass = clazz.getGenericSuperclass();
463         if (!(genericSuperclass instanceof ParameterizedType)) {
464             return resolveGenericClass(clazz.getSuperclass(), genericTypeArgIndex);
465         }
466 
467         ParameterizedType parameterizedType = (ParameterizedType) genericSuperclass;
468         Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
469         if (actualTypeArguments.length != 2) {
470             return resolveGenericClass(clazz.getSuperclass(), genericTypeArgIndex);
471         }
472 
473         Type listenerType = actualTypeArguments[genericTypeArgIndex];
474         if (!(listenerType instanceof Class<?>)) {
475             return resolveGenericClass(clazz.getSuperclass(), genericTypeArgIndex);
476         }
477 
478         return (Class<T>) listenerType;
479     }

481     private static class EventMethodResolver implements Supplier<List<Method>> {
482 
483         private String[] methodNames;
484         private Supplier<Class<?>> listenerClassSupplier;
485         private Supplier<Class<?>> eventClassSupplier;
486 
487         public EventMethodResolver(String[] methodNames, Supplier<Class<?>> listenerClassSupplier, Supplier<Class<?>> eventClassSupplier) {
488             this.methodNames = methodNames;
489             this.listenerClassSupplier = listenerClassSupplier;
490             this.eventClassSupplier = eventClassSupplier;
491         }
492 
493         @Override
494         public List<Method> get() {
495             Class<?> listenerClass = listenerClassSupplier.get();
496             Class<?> eventClass = this.eventClassSupplier.get();
497             if (methodNames.length == 0) {
498                 return resolveListenerMethods(listenerClass, eventClass);
499             } else {
500                 return resolveListenerMethodsByName(methodNames, listenerClass, eventClass);
501             }
502         }
503 
504         private List<Method> resolveListenerMethods(Class<?> clazz, Class<?> eventObjectClass) {
505             Class<?>[] eventMethodParams = new Class<?>[]{eventObjectClass};
506             List<Method> listenerMethodNames = stream(clazz.getDeclaredMethods())
507                     .filter(m -> Arrays.equals(m.getParameterTypes(), eventMethodParams))
508                     .collect(toList());
509 
510             Class<?>[] superinterfaces = clazz.getInterfaces();
511 
512             List<Method> superMethodNames = stream(superinterfaces)
513                     .map(si -> resolveListenerMethods(si, eventObjectClass))
514                     .flatMap(List::stream)
515                     .collect(toList());
516 
517             listenerMethodNames.addAll(superMethodNames);
518             return listenerMethodNames;
519         }
520 
521         private List<Method> resolveListenerMethodsByName(String[] methodNames, Class<?> listenerClass, Class<?> eventObjectClass) {
522             List<Method> eventMethods = new ArrayList<>();
523             for (String methodName : methodNames) {
524                 try {
525                     Method eventMethod = listenerClass.getDeclaredMethod(methodName, eventObjectClass);
526                     eventMethods.add(eventMethod);
527                 } catch (NoSuchMethodException | SecurityException e) {
528                     throw new IllegalArgumentException(e);
529                 }
530             }
531             return eventMethods;
532         }
533     }
