37      /**
38       * Traverses the graph using the breadth first traversal strategy. Starting at
39       * the given {@link Node} and applies the given {@link Consumer} on every
40       * {@link Node} .
41       *
42       * @param start         the {@link Node} to start the traversal from.
43       * @param nodeProcessor the closure that should be applied to every node.
44       * @since 1.2.0;
45       */
46      public static void traverseBreadthFirst(Node start, Consumer<Node> nodeProcessor) {
47          Iterator<Node> objectGraphIterator = new BreadthFirstNodeIterator(start);
48          forAllDo(objectGraphIterator, nodeProcessor);
49      }
50  
51      /**
52       * Traverses the graph using the depth first traversal strategy. Starting at the
53       * given {@link Node} and applies the given {@link Consumer} on every
54       * {@link Node} .
55       *
56       * @param start         the {@link Node} to start the traversal from.
57       * @param nodeProcessor the closure that should be applied to every node.
58       * @since 1.0.0;
59       */
60      public static void traverseDepthFirst(Node start, Consumer<Node> nodeProcessor) {
61          DepthFirstNodeIterator objectGraphTransformer = new DepthFirstNodeIterator(start);
62          forAllDo(objectGraphTransformer, nodeProcessor);
63      }
64  
65      static <T> void forAllDo(Iterator<T> iterator, Consumer<T> closure) {
66          while (iterator.hasNext()) {
67              T next = iterator.next();
68              closure.accept(next);
69          }
70      }

72      public enum NodeIterateStrategy {
73          DEPTH_FIRST {
74              @Override
75              public Iterator<Node> iterator(Node startNode) {
76                  return new DepthFirstNodeIterator(startNode);
77              }
78          }, BREADTH_FIRST {
79              @Override
80              public Iterator<Node> iterator(Node startNode) {
81                  return new BreadthFirstNodeIterator(startNode);
82              }
83          };

88      /**
89       * Creates a predicated {@link Node} iterator that iterates the {@link Node} s
90       * per Predicate that is specified using the {@link NodeIterateStrategy}.
91       * <p>
92       * Take the following node structure for example
93       *
94       * <pre>
95       *                   A
96       *      +------------+------------+
97       *      B            C            D
98       * +----+----+       |        +---+------------+
99       * E    F    G       H        I   J            K
100      *                                      +------+------+
101      *                                      L      M      N
102      * </pre>
103      * <p>
104      * If we assume that we have 3 {@link Predicate}s
105      * <ul>
106      * <li>The 1. {@link Predicate} matches {@link Node}s A,C,H,D,K</li>
107      * <li>The 2. {@link Predicate} matches {@link Node}s A,B,E,F</li>
108      * <li>The 3. {@link Predicate} matches {@link Node}s D,J,M,N</li>
109      * </ul>
110      * and we construct a per predicated node iterator using
111      *
112      * <pre>
113      * {@link GraphFacade#perPredicateNodeIterator(NodeIterateStrategy, Node, Predicate...) GraphFacade.perPredicatedNodeIterator(BREADTH_FIRST, startNodeA, pred1, pred2, pred3)};
114      * </pre>
115      * <p>
116      * The resulting iterator will iterate the node structure using a breadth first
117      * strategy for every {@link Predicate} starting at startNodeA. <br/>
118      * The result will be:
119      *
120      * <pre>
121      * breadth first  breadth first   breadth first
122      *   matching       matching       matching
123      *    pred1          pred2           pred3
124      * +-----------+   +-------+       +-------+
125      *   A C D H K      A B E F         D J M N
126      *
127      * iterate order -->
128      * </pre>
129      *
130      * </p>
131      *
132      * @param nodeIterateStrategy
133      * @param startNode
134      * @param nodeIterateOrderPredicates
135      * @return
136      */
137     @SuppressWarnings("unchecked")
138     public static Iterator<Node> perPredicateNodeIterator(NodeIterateStrategy nodeIterateStrategy, Node startNode, Predicate<Node>... nodeIterateOrderPredicates) {
139         class FilteredIterator<E> implements Iterator<E> {
140 
141             private final Iterator<E> filtered;
142 
143             public FilteredIterator(Iterator<E> iterator, Predicate<E> filter) {
144                 filtered = stream(spliteratorUnknownSize(iterator, ORDERED), false).filter(filter).iterator();
145             }
146 
147             @Override
148             public boolean hasNext() {
149                 return filtered.hasNext();
150             }
151 
152             @Override
153             public E next() {
154                 return filtered.next();
155             }
156         }
157 
158         class ChainedIterator<E> implements Iterator<E> {
159 
160             private Iterator<Iterator<E>> chain;
161             private Iterator<E> current = Collections.emptyIterator();
162 
163             @SafeVarargs
164             public ChainedIterator(Iterator<E>... iterators) {
165                 this(asList(iterators));
166             }
167 
168             public ChainedIterator(Collection<Iterator<E>> iterators) {
169                 chain = iterators.iterator();
170             }
171 
172             @Override
173             public boolean hasNext() {
174                 while (!current.hasNext() && chain.hasNext()) {
175                     current = chain.next();
176                 }
177 
178                 return current.hasNext();
179             }
180 
181             @Override
182             public E next() {
183                 if (!hasNext()) {
184                     throw new NoSuchElementException();
185                 }
186                 return current.next();
187             }
188         }
189 
190 
191         List<Iterator<Node>> iterators = new ArrayList<>();
192         for (int i = 0; i < nodeIterateOrderPredicates.length; i++) {
193             Predicate<Node> predicate = nodeIterateOrderPredicates[i];
194             Iterator<Node> nodeStrategyIterator = nodeIterateStrategy.iterator(startNode);
195             Iterator<Node> predicateFilterIterator = new FilteredIterator<>(nodeStrategyIterator, predicate);
196             iterators.add(predicateFilterIterator);
197         }
198         return new ChainedIterator<>(iterators);
199     }
